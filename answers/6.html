Palindrome program

        
            function palindrome(str){
                let first = 0;
                let last = str.length-1;

                while(first < last){
                    if(str[first] !== str[last]){
                        return false;
                    }
                    first++;
                    last--;
                }
                return true
            }
            palindrome('racecar')

            function pallindrome(str){
                let len = str.length
                let mid = Math.floor(len/2);

                for(let i =0; i< mid; i++){
                    if(str[i] !== str[len-1-i]){
                        return false;
                    }
                }
                return true
            }
            pallindrome('racecar')
